public String addCountry(Country c) {
            Session session=null;
            Transaction tx=null;
            String msg=null;

            try {
            session=factory.openSession();
            tx=session.beginTransaction();
            session.persist(c);
            tx.commit();

            msg="Country Added Sucessfully..";
            }catch(Exception e) {
                if(tx!=null) {
                    tx.rollback();
                }

                e.printStackTrace();
            }finally {
                if(session!=null) {
                    session.close();
                }
            }
            return msg;
           }
